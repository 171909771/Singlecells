##########################reference合并注释
```r
annotation_columns <- c("scinamarker", "panglao", "tmp")

get_consensus_label <- function(labels){
  labels <- labels[labels != "ambiguous"]
  if (length(labels) == 0) {return("ambiguous")}
  freq <- table(labels)
  label <- names(freq)[which(freq == max(freq))]
  if (length(label) > 1) {return("ambiguous")}
  return(label)
}

annotation1@meta.data$identical=apply(annotation1@meta.data[,annotation_columns], 1, get_consensus_label)
  
# 查看 
DimPlot(annotation1, reduction = "tsne", label = T, group.by = 'identical')


annotation1=subset(annotation1,identical != 'neural progenitor cells' & identical != 'ambiguous')
annotation1@meta.data[["identical"]]=factor(annotation1@meta.data[["identical"]],levels=unique(annotation1@meta.data[["identical"]]))



rm(list=ls()[ls()!='obj.merge'])
gc()

saveRDS(annotation1,file = 'annotation2.finalcluster.rds')
```
